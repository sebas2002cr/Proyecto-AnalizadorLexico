
function factorial:int(numero:int) {
	local resultado:int := 1 |
	do {
		resultado := resultado * numero |
		--numero |
	} until(numero = 1) |
	return resultado |
}

function main() {
	print(cuadrado(4)) |
	print("\n") |
	print(potencia(4, 3)) |
	print("\n") |
	print(factorial(6)) |
	print("\n") |
	print(indice(2)) |
	print("\n") |
	prueba() |
}

function cuadrado:int(numero:int) {
	return numero * numero |
}

function potencia:int(base:int, exponente:int) {
	local resultado:int := base |
	for(i:int := 1, i < exponente, ++i) {
		resultado := resultado * base |
	}
	return resultado |
}

function indice:string(i:int) {
	if(i = 0) {
		return "es 0" |
	} elif (i = 1) {
		return "es 1" |
	} else {
		return "es otro" |
	}
}

function prueba:char() {
	local contador:int |
	for(contador := 0, contador < 10, ++contador) {
		print(contador) |
		print(' ') |
		if(contador = 5) {
			break |
		}
	}
	return '1' |
}
